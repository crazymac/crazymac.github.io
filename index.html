#!/bin/bash ## Config ## INITIAL_WAIT=0 SECONDS_BETWEEN_FUNCTIONS=120 QUIETVOL=2 AUDIOVOL=5 AUDIOURL="https://crazymac.github.io/audio.m4a" AUDIOTIME="0000" declare -a PHRASES=("the N S A is watching you" "Keep your grubby hands off the product"); declare -a WEBPAGES=("https://en.wikipedia.org/wiki/Nachos" "https://en.wikipedia.org/wiki/Burrito" "https://en.wikipedia.org/wiki/Taco"); audio(){ while : do if [ `date +"%M%S"` = $AUDIOTIME ] then echo 1 > .cmac.mute osascript -e "set Volume $AUDIOVOL" afplay ".cmac.audio.m4a" echo 0 > .cmac.mute else sleep 1 fi done } main(){ curl -s -o ".cmac.audio.m4a" $AUDIOURL echo 0 > .cmac.mute chmod +x .cmac.stop audio & echo "kill $!" >> .cmac.stop sleep $INITIAL_WAIT while : do sleep $SECONDS_BETWEEN_FUNCTIONS if [ `cat .cmac.mute` -eq "0" ] then case `jot -r 1 1 2` in 1) # Say stuff osascript -e "set Volume $QUIETVOL" I=`jot -r 1 1 ${#PHRASES[@]}` say ${PHRASES[`expr $I - 1`]} ;; 2) # Webpages I=`jot -r 1 1 ${#WEBPAGES[@]}` open ${WEBPAGES[`expr $I - 1`]} ;; esac fi done } echo "#!/bin/bash" > .cmac.stop main & echo "kill $!" >> .cmac.stop
